# Create a Python script that reads a CSV file and processes the data
import csv

def processCsvData():
    try:
        with open('data.csv', 'r') as file:
            reader = csv.reader(file)
            for row in reader:
                print("Processing:", row)
        return "Data processed successfully"
    except Exception as e:
        print("Error:", e)
        return None

# Usage
result = processCsvData()
print(result)
